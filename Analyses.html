function dropoutliers(df, α = thresh)
    degreesOF = size(df)[2]
    covMatrix = StatsBase.cov(Matrix(df))
    μ = reshape(mean(Matrix(df), dims = 1),:)
    thresh = sqrt(quantile(Chisq(degreesOF), 1 - α))
    [mahalanobis(Vector(r), μ, covMatrix^(-1)) for r in eachrow(df)] .< thresh
end

function myheatmap(M;title = "", xlabel = "", ylabel = "", other...)
    M1 = floor.(1000(M./sum(M,dims=1)))./10
    M = convert.(Int,M)
    reg(t) = "$(t[1])% ($(t[2]))"
    annotations = reg.(collect(zip(M1,M)))
    Seaborn.set(rc=Dict("figure.figsize"=>(11.7,8.27)))
    fig = Seaborn.heatmap(M1./100, annot = annotations, cmap = "magma", fmt = "s"; other...)
    fig.set(xlabel = xlabel, ylabel = ylabel)
    fig.set_title(title, fontsize = 20, y = 1.02)
    fig
end

function getAnalysis(;dep, indep, conf = vConf, subsample = :total)
    # uses dataframe `data` already loaded in R environment
    names1 = ["Residuals Shapiro-Wilk's W","p-value"]
    
        
    # linear model fitting
    if subsample == :total
        reval("""
            model <- lm(`$dep` ~ $indep + $(join(conf," + ")), data = data)        
        """);
    else
        reval("""
            model <- lm(`$dep` ~ $indep + $(join(conf," + ")), data = FEPdata)
        """);        
    end

    # residuals' normality test
    shapirowilk = hcat(
        DataFrame(DepVarAnalysis = dep),
        DataFrame(IndepVar = indep),
        rename(DataFrame(rcopy(R"shapiro.test(model$residuals)"))[:,1:2], names1)
    )
    
    # emmeans (forse bisogna eliminare l'aggiustamento e farlo dopo)
    reval("""
       res <- emmeans(model, specs = pairwise ~ $indep, adjust = "none")
    """)
    
    names2 = ["value","emmean"]
    resemmeans = rename(rcopy(R"as.data.frame(res$emmeans)")[:,[1,2]], names2)
    emmeans = hcat(
        DataFrame(DepVarAnalysis = repeat([dep],size(resemmeans)[1])),
        DataFrame(Factor = repeat([indep],size(resemmeans)[1])),
        resemmeans
    )
    
    names3 = ["contrast","estimate","p-value"]
    rescontrast = rename(rcopy(R"as.data.frame(res$contrast)")[:,[1,2,6]], names3)
    contrast = hcat(DataFrame(DepVarAnalysis = repeat([dep],size(rescontrast)[1])),rescontrast)
    
    (
        shapirowilk = shapirowilk,
        emmeans = emmeans,
        contrast = contrast
    )
end
